    \section{Реализация метода bootstrap}

    Метод bootstrap: \\
    Имеется выборка $X_1, \ldots, X_n$ из $X$ \\
    $\alpha -$ уровень доверия \\
    Алгоритм вычисления доверительного интервала параметра $\theta$ \\
    \begin{enumerate}
        \item Найти оценку параметра $\widehat{\theta}_n$
        \item Сгенерировать $k$ новых выборок $X_1^i, \ldots, X_n^i, ~~ i = \overline{1, k}, ~~ X_j^i -$ случайный элемент исходной выборки
        \item $\theta_i^* = \widehat{\theta}_n (X_1^i, \ldots, X_n^i), ~~ i = \overline{1, k}$
        \item Доверительный интервал -- $\displaystyle \left( \theta_{\left\lfloor {\textstyle \frac{1 - \alpha}{2}} k \right\rfloor}^*; \theta_{\left\lceil {\textstyle \frac{1 + \alpha}{2}} k \right\rceil}^* \right)$
    \end{enumerate}

    Моя реализация:
    \begin{minted}{python}
        def bootstrap(X_n: List[float], theta_n: Callable[[List[float]], float], alpha: float = 0.05) -> Tuple[float, float]:
            k = 1000
            theta_i = [0] * k
            
            for i in range(k):
                X_i = random.choices(X_n, k=len(X_n))
                theta_i[i] = theta_n(X_i)
            theta_i.sort()
            
            left = math.floor((1 - alpha) / 2 * k)
            right = math.ceil((1 + alpha) / 2 * k)
            
            return theta_i[left], theta_i[right]
    \end{minted}

    \subsection{Равномерное распределение}

    Оценим параметры равномерного распределения методом максимального правдоподобия: \\
    $\displaystyle
    X \sim U(a, b) \\
    X_1, \ldots, X_n - \text{ выборка из } X \\
    F_X \in \{F(x, a, b) ~|~ a, b \in \mathbb{R}, b \geq a\} \\
    f_X(x) = \frac{I(x \in [a; b])}{(b - a)^n} \\
    L(a, b, X_1, \ldots, X_n) = \prod_{i = 1}^n \frac{I(X_i \in [a; b])}{(b - a)^n} = \frac{1}{(b - a)^n} \prod_{i = 1}^n I(X_i \in [a; b]) \\
    \hat{a}_n = \arg\max_{a \leq b} \frac{1}{(b - a)^n} \prod_{i = 1}^n I(X_i \in [a; b]) = \arg\max_{a \leq X_{\min}} \frac{1}{(b - a)^n} = X_{\min} \\
    \hat{b}_n = \arg\max_{b \geq a} \frac{1}{(b - a)^n} \prod_{i = 1}^n I(X_i \in [a; b]) = \arg\max_{b \geq X_{\max}} \frac{1}{(b - a)^n} = X_{\max} \\
    \text{Оценка параметра } \hat{a}_n = X_{\min} \\
    \text{Оценка параметра } \hat{b}_n = X_{\max} \\
    $

    \subsection{Распределение Бернулли}

    Оценим параметр биномиального распределения методом максимального правдоподобия: \\
    $\displaystyle
    X \sim Bernoulli(p): X_1, X_2, \ldots, X_n \\
    F_X \in \{F_X(x, p) | p \in [0; 1] \} \\
    p_X(x) = p^x (1 - p)^{1 - x} \\
    L(p, X_1, \ldots, X_n) = \prod_{i = 1}^n p^{X_i} (1 - p)^{1 - X_i} = p^{\scriptstyle \sum\limits_{i = 1}^n X_i} (1 - p)^{\scriptstyle n - \sum\limits_{i = 1}^n X_i} \\
    \ln L(p, X_1, \ldots, X_n) = \ln \left( p^{\scriptstyle \sum\limits_{i = 1}^n X_i} (1 - p)^{\scriptstyle n - \sum\limits_{i = 1}^n X_i} \right) = \\ \sum_{i = 1}^n X_i \ln p + \left( n - \sum_{i = 1}^n X_i \right) \ln(1 - p) \\
    \hat{p}_n = \arg\max_{p \in [0; 1]} \ln L(p, X_1, \ldots, X_n) \\
    $
    Находим максимума по аргументу: \\
    $\displaystyle
    \frac{d\ln L}{dp} = \frac{\displaystyle \sum_{i = 1}^n X_i}{p} + \frac{\displaystyle \sum_{i = 1}^n X_i - n}{1 - p} = 0 \\
    (1 - p) \sum_{i = 1}^n X_i + \left( \sum_{i = 1}^n X_i - n \right) p = 0 \\
    pn = \sum_{i = 1}^n X_i \\
    p = \sum_{i = 1}^n \frac{X_i}{n} \\
    $
    Проверка максимума: \\
    $\displaystyle
    \frac{d^2 \ln L}{dp^2} = \underbrace{-\frac{\displaystyle \sum_{i = 1}^n X_i}{p^2}}_{\displaystyle \leq 0} + \underbrace{\frac{\displaystyle \sum_{i = 1}^n X_i - n}{(1 - p)^2}}_{\displaystyle \leq 0} \leq 0 \\
    $
    Оценка параметра $\displaystyle \hat{p}_n = \sum_{i = 1}^n \frac{X_i}{n} = \overline{X}$

    Код:

    \subsection{Биномиальное распределение}

    Оценим параметры биномиального распределения методом моментов: \\
    $\displaystyle
    X \sim Bin(k, p) \\
    X_1, \ldots, X_n - \text{ выборка из } X \\
    F_X \in \{F_X(x, k, p) ~|~ p \in [0; 1], k \in \mathbb{N} \} \\
    \begin{cases}
        E(X) = \overline{X} \\
        D(X) = \overline{X^2} - \overline{X}^2
    \end{cases} \Leftrightarrow \begin{cases}
        kp = \overline{X} \\
        kp(1 - p) = \overline{X^2} - \overline{X}^2
    \end{cases} \Leftrightarrow \begin{cases}
        \displaystyle p = \frac{\overline{X} - \overline{X^2} + \overline{X}^2}{\overline{X}} \\
        \displaystyle k = \frac{\overline{X}^2}{\overline{X} - \overline{X^2} + \overline{X}^2}
    \end{cases} \\
    \text{Оценка параметра } \hat{p}_n = \frac{\overline{X} - \overline{X^2} + \overline{X}^2}{\overline{X}} \\
    \text{Оценка параметра } \hat{k}_n = \frac{\overline{X}^2}{\overline{X} - \overline{X^2} + \overline{X}^2}
    $

    Код:
    \begin{minted}{python}
        def p_n_Bin(X_n: List[float]) -> float:
            X_mean = sum(X_n) / len(X_n)
            X_2_mean = sum(X ** 2 for X in X_n) / len(X_n)

            return (X_mean - X_2_mean + X_mean ** 2) / X_mean

        def k_n_Bin(X_n: List[float]) -> float:
            X_mean = sum(X_n) / len(X_n)
            X_2_mean = sum(X ** 2 for X in X_n) / len(X_n)

            return X_mean ** 2 / (X_mean - X_2_mean + X_mean ** 2)

    \subsection{Нормальное распределение}

    Оценим параметры нормального распределения методом моментов: \\
    $\displaystyle
    X \sim N(\mu, \sigma^2) \\
    X_1, \ldots, X_n - \text{ выборка из } X \\
    F_X \in \{ F_X(x, \mu, \sigma^2) ~|~ \mu, \sigma \in \mathbb{R} \} \\
    \begin{cases}
        E(X) = \overline{X} \\
        D(X) = \overline{X^2} - \overline{X}^2
    \end{cases} \Leftrightarrow \begin{cases}
        \mu = \overline{X} \\
        \sigma^2 = \overline{X^2} - \overline{X}^2
    \end{cases} \Leftrightarrow \begin{cases}
        \mu = \overline{X} \\
        \sigma = \sqrt{\overline{X^2} - \overline{X}^2}
    \end{cases} \\
    \text{Оценка параметра } \hat{\mu}_n = \overline{X} \\
    \text{Оценка параметра } \hat{\sigma}_n = \sqrt{\overline{X^2} - \overline{X}^2}
    $
