    $\xi_1, \xi_2, \ldots, \xi_{12}$ независимые, $\xi_i \sim \Uniform(0; 13) ~~ i = \overline{1, 12}$, тогда: \\
    $\displaystyle
    \forall i = \overline{1, 12} ~~ F_{\xi}(t) = F_{\xi_i}(t) = \begin{cases}
        0, & x < 0 \\
        \displaystyle \frac{t}{13}, & x \in [0; 13] \\
        1, & x > 13
    \end{cases} \\

    1 - \prod_{i = 1}^{12} P(\xi_i > x) =
    1 - \prod_{i = 1}^{12} (1 - P(\xi_i \leq x)) =
    1 - \prod_{i = 1}^{12} (1 - F_{\xi_i}(x)) = \\
    \begin{cases}

        \displaystyle 1 - \prod_{i = 1}^{12} \left( 1 - \frac{x}{13} \right), & x \in [0; 13] \\
        \displaystyle 1 - \prod_{i = 1}^{12} (1 - 1), & x > 13
    \end{cases} =
    \begin{cases}
        0, & x < 0 \\
        \displaystyle 1 - \left( 1 - \frac{x}{13} \right)^{12}, & x \in [0; 13] \\
        1, & x > 13
    \end{cases} \\
    F_X(x) = 1 - \left( 1 - \frac{x}{13} \right)^{12} I(x \in [0; 13]) + I(x > 13) \\
    f_X(x) = \frac{d}{dx} F_X(x) = \frac{12}{13} \left( 1 - \frac{x}{13} \right)^{11} \cdot I(x \in [0; 13]) \\\\
    Y = \max(\xi_1, \xi_2, \ldots, \xi_{12}) \Rightarrow Y \leq x \Leftrightarrow \forall i \in \{1, 2, \ldots, 12\} ~~ \xi_i \leq x \Rightarrow \\

        \displaystyle \left( \frac{y}{13} \right)^{12}, & y \in [0; 13] \\
        1, & y > 13
    \end{cases} \\
    F_Y(y) = \left( \frac{y}{13} \right)^{12} I(y \in [0; 13]) + I(y > 13) \\
    f_Y(y) = \frac{d}{dy} F_Y(y) = \frac{12}{13} \left( \frac{y}{13} \right)^{11} I(y \in [0; 13]) \\\\
    a) E(X) = \int_{\mathbb{R}} x f_X(x) \,dx = \int_{-\infty}^{+\infty} x \frac{12}{13} \left( 1 - \frac{x}{13} \right)^{11} I(x \in [0; 13]) \,dx = \\
    \frac{12}{13} \int_{0}^{13} x \left( 1 - \frac{x}{13} \right)^{11} \,dx = 
    \left| \begin{array}{ll}
        u = x & du = dx \\
        \displaystyle dv = \left( 1 - \frac{x}{13} \right)^{11} dx & \displaystyle v = -\frac{13}{12} \left( 1 - \frac{x}{13} \right)^{12}
    \end{array} \right| =
    -x \left( 1 - \frac{x}{13} \right)^{12} \Big|_{0}^{13} + \int_{0}^{13} \left( 1 - \frac{x}{13} \right)^{12} \,dx = 
    -\left( 1 - \frac{x}{13} \right)^{13} \Big|_{0}^{13} =

    \frac{12}{13} \int_{0}^{13} y \left( \frac{y}{13} \right)^{11} \,dy = 
    \frac{12}{13^{12}} \int_{0}^{13} y^{12} \,dy =
    \frac{12}{13^{13}} \cdot y^{13} \Big|_{0}^{13} =

    E(X^2) = \int_{\mathbb{R}} x^2 f_X(x) \,dx = \int_{-\infty}^{+\infty} x^2 \frac{12}{13} \left( 1 - \frac{x}{13} \right)^{11} I(x \in [0; 13]) \,dx = \\
    \frac{12}{13} \int_{0}^{13} x^2 \left( 1 - \frac{x}{13} \right)^{11} \,dx =

        \displaystyle dv = \left( 1 - \frac{x}{13} \right)^{11} dx & \displaystyle v = -\frac{13}{12} \left( 1 - \frac{x}{13} \right)^{12}
    \end{array} \right| = \\
    -x^2 \left( 1 - \frac{x}{13} \right)^{12} \Big|_{0}^{13} + 2 \int_{0}^{13} x \left( 1 - \frac{x}{13} \right)^{12} \,dx =
    2 \int_{0}^{13} x \left( 1 - \frac{x}{13} \right)^{12} \,dx =
    \left| \begin{array}{ll}

        \displaystyle dv = \left( 1 - \frac{x}{13} \right)^{12} dx & \displaystyle v = -\left( 1 - \frac{x}{13} \right)^{13}
    \end{array} \right| =
    -2x \left( 1 - \frac{x}{13} \right)^{13} \Big|_{0}^{13} +\\ 2 \int_{0}^{13} \left( 1 - \frac{x}{13} \right)^{13} \,dx =
    -2 \cdot \frac{13}{14} \left( 1 - \frac{x}{13} \right)^{14} \Big|_{0}^{13} =
    -\frac{13}{7} \\
    \circledeq \frac{13}{7} - 1 = \frac{6}{7} \\
    D(Y) = E(Y^2) - E^2(Y) \circledeq \\
    E(Y^2) = \int_{\mathbb{R}} y^2 f_Y(y) \,dy = \int_{-\infty}^{+\infty} y^2 \frac{12}{13} \left( \frac{y}{13} \right)^{11} I(y \in [0; 13]) \,dy = \\
    \frac{12}{13} \int_{0}^{13} y^2 \left( \frac{y}{13} \right)^{11} \,dy = \frac{12}{13^{12}} \cdot \frac{y^{14}}{14} \Big|_{0}^{13} =
    \frac{1014}{7} \\
    \circledeq \frac{1014}{7} - 144 = \frac{6}{7} \\
    \Cov(X, 3Y) = 3E(XY) - 3E(X)E(Y) \circledeq \\
    F_{X, Y}(x, y) = P(X \leq x \cap Y \leq y) =
    P\left( \left( \bigcup_{i = 1}^{12} \xi_i \leq x \right) \cap \left( \bigcap_{i = 1}^{12} \xi_i \leq y \right) \right) =
    P\left( \bigcup_{i = 1}^{12} \left( \xi_i \leq x \cap \left( \bigcap_{j = 1}^{12} \xi_j \leq y \right) \right) \right) =
    \begin{cases}
        \displaystyle P\left( \bigcup_{i = 1}^{12} \left( \xi_i \leq x \cap \left( \bigcap_{j = 1}^{12} \xi_j \leq y \right) \right) \right), & x < y \\
        \displaystyle P\left( \bigcup_{i = 1}^{12} \left( \bigcap_{j = 1}^{12} \xi_j \leq y \right) \right), & x \geq y
    \end{cases} \\=
    \begin{cases}
        \displaystyle \sum_{k = 1}^{12} (-1)^{k + 1} \sum_{0 \leq i_1 < i_2 < \ldots < i_k < 12} P\left( \bigcap_{i = 1}^{k} \left( \xi_i \leq x \cap \left( \bigcap_{j = 1}^{12} \xi_j \leq y \right) \right) \right), & x < y \\
        \displaystyle P\left( \bigcap_{j = 1}^{12} \xi_j \leq y \right), & x \geq y
    \end{cases} =
    \begin{cases}
        \displaystyle \sum_{k = 1}^{12} (-1)^{k + 1} C_{12}^{k} P\left( \bigcap_{i = 1}^{k} \left( \xi_i \leq x \cap \left( \bigcap_{j = 1}^{12} \xi_j \leq y \right) \right) \right), & x < y \\
        \displaystyle P\left( \bigcap_{j = 1}^{12} \xi_j \leq y \right), & x \geq y
    \end{cases} = \\
    \begin{cases}
        \displaystyle \sum_{k = 1}^{12} (-1)^{k + 1} C_{12}^{k} P\left( \bigcap_{i = 1}^{k} \left( \xi_i \leq x \cap \left( \bigcap_{j = 1}^{12} \xi_j \leq y \right) \right) \right), & x < y \\
        \displaystyle \prod_{j = 1}^{12} P(\xi_j \leq y), & x \geq y
    \end{cases} = \\
    \begin{cases}

        \displaystyle F_Y(y), & x \geq y

        \displaystyle \left( \frac{y}{13} \right)^{12}, & 0 \leq y \leq x \leq 13 \\
        \displaystyle 1 - \left( 1 - \frac{x}{13} \right)^{12}, & 0 \leq x \leq 13 < y \\
