        \displaystyle 1, & \displaystyle \left( x > \frac{\pi}{2} \right) \land \left( y > \frac{\pi}{2} \right)

    \begin{cases}

        \displaystyle -\frac{1}{2} (\sin(u + y) - \sin u) \Big|_{0}^{x}, & \displaystyle \left( 0 \leq x \leq \frac{\pi}{2} \right) \land \left( 0 \leq y \leq \frac{\pi}{2} \right) \\

    2) f_X(x) = \int_{\mathbb{R}} f_{X, Y}(x, y) \,dy = 
    \int_{-\infty}^{+\infty} \frac{1}{2} \sin(x + y) \cdot I\left( (x, y) \in \left[ 0; \frac{\pi}{2} \right]^2 \right) \,dy =
    \frac{1}{2} \cdot I\left( x \in \left[ 0; \frac{\pi}{2} \right] \right) \int_{0}^{\textstyle \frac{\pi}{2}} \sin(x + y) \,dy =
    -\frac{1}{2} \cos(x + y) \Big|_{0}^{\textstyle \frac{\pi}{2}} \cdot I\left( x \in \left[ 0; \frac{\pi}{2} \right] \right) =
    -\frac{1}{2} \left( \cos \left( x + \frac{\pi}{2} \right) - \cos x \right) \cdot I\left( x \in \left[ 0; \frac{\pi}{2} \right] \right) =
    \frac{1}{2} (\sin x + \cos x) \cdot I\left( x \in \left[ 0; \frac{\pi}{2} \right] \right) \\\\
    3) f_{X|Y}(x|y) = \frac{f_{X, Y}(x, y)}{f_Y(y)} =
    \frac{\displaystyle \frac{1}{2} \sin(x + y) \cdot I\left( (x, y) \in \left[ 0; \frac{\pi}{2} \right]^2 \right)}{\displaystyle \frac{1}{2} (\sin y + \cos y) \cdot I\left( y \in \left[ 0; \frac{\pi}{2} \right] \right)} =
    \frac{\sin(x + y)}{(\sin y + \cos y)} \cdot I\left( (x, y) \in \left[ 0; \frac{\pi}{2} \right]^2 \right) \\\\
    4) E(X) = \int_{\mathbb{R}} x f_X(x) \,dx = 
    \int_{-\infty}^{+\infty} x \frac{\sin x + \cos x}{2} \cdot I\left( x \in \left[ 0; \frac{\pi}{2} \right] \right) \,dx =
    \frac{1}{2} \int_{0}^{\textstyle \frac{\pi}{2}} x (\sin x + \cos x) \,dx =
    \left| \begin{array}{ll}
        u = x & du = dx \\
        dv = (\sin x + \cos x) dx & v = \sin x - \cos x
    \end{array} \right| =
    \frac{1}{2} x (\sin x - \cos x) \Big|_{0}^{\textstyle \frac{\pi}{2}} - \frac{1}{2} \int_{0}^{\textstyle \frac{\pi}{2}} (\sin x - \cos x) \,dx =
    \frac{1}{2} \cdot \frac{\pi}{2} - \frac{1}{2} (-\cos x - \sin x) \Big|_{0}^{\textstyle \frac{\pi}{2}} =
    \frac{\pi}{4} + \frac{1}{2} -\frac{1}{2} = \frac{\pi}{4} \\
    D(X) = E(X^2) - E^2(X) \circledeq \\

    6) \Cov(X, Y) = E(XY) - E(X)E(Y) \circledeq \\

    \int_{-\infty}^{+\infty} \int_{-\infty}^{+\infty} xy \frac{1}{2} \sin(x + y) \cdot I\left( (x, y) \in \left[ 0; \frac{\pi}{2} \right]^2 \right) \,dxdy =
    \frac{1}{2} \int_{0}^{\textstyle \frac{\pi}{2}} x \,dx \int_{0}^{\textstyle \frac{\pi}{2}} y \sin(x + y) \,dy =

        dv = \sin(x + y) dy & v = -\cos(x + y)
    \end{array} \right| =

    \frac{1}{2} \int_{0}^{\textstyle \frac{\pi}{2}} x \left( \left( \frac{\pi}{2} - 1 \right) \sin x + \cos x \right) \,dx \\=

        u = x & du = dx \\

    \frac{1}{2} \left( x \left( \left( \frac{\pi}{2} - 1 \right) \sin x + \cos x \right) \Big|_{0}^{\textstyle \frac{\pi}{2}} - \int_{0}^{\textstyle \frac{\pi}{2}} \left( \left( \frac{\pi}{2} - 1 \right) \cos x + \sin x \right) \,dx \right) = \\
    \frac{1}{2} \left( \frac{\pi}{2} - \left( \left( 1 - \frac{\pi}{2} \right) \sin x + \cos x \right) \Big|_{0}^{\textstyle \frac{\pi}{2}} \right) =
    \frac{1}{2} \left( \frac{\pi}{2} - 1 + \frac{\pi}{2} - 1 \right) =
    \frac{1}{2} \cdot 2 =
    1 \\
    \circledeq 1 - \frac{\pi^2}{16} \\\\
    7) \rho(X, Y) = \frac{\Cov(X, Y)}{\sigma_X \sigma_Y} =
    \frac{\displaystyle 1 - \frac{\pi^2}{16}}{\displaystyle \sqrt{\left( \frac{\pi^2}{16} + \frac{\pi}{2} - 2 \right)^2}} =
